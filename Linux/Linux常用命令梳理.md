```shell
#查看linux IP信息
	ifconfig

#查看指定目录内容
	ls
		-a 所有文件包括隐藏   在linux中以.开始的名字都是隐藏的
		-l 文件完整信息 包括权限 大小 日期   ls -l = ll
#切换目录
	cd

#当前所在目录路径
	pwd

#创建文件夹
	mkdir
		-p  父目录不存在 自动创建

#创建一个空文件
	touch
	
#复制 移动
	cp
		-r 移动文件夹使用  递归
	mv
		可以用于重命名
#删除
	rm
		-f 强制 不提示
		-r 递归
#查看文件内容
	cat  适合小文件
	more less 翻页
	head
	tail
		-f  实时更新文件追加变化的内容
#其他
	|     管道
	echo  输出  print
	>  输出重定向  覆盖
	>> 输出重定向  追加
	
#搜索
	find
	grep  在管道中搜索指定的字符串
	which 用于查询命令安装路径 位置

#软链接
	ln -s 目标的绝对路径  软链接名（快捷方式）
	
#打包 压缩
	tar
		-zcvf  c:create  z:gzip
		
		-zxvf  x:execute
		
		-C    指定解决到那个目录下 如果不指定 默认解压到当前
		
#文件系统特殊符号
	/  根目录
	.  在路径中出现  表示是当前  在文件名字中出现  表示的是隐藏
	.. 上一级
	~  用户的家目录  root用户是linux超级管理员
	
#混淆概念
	当前路径
	相对路径
	绝对路径
	
#vim编辑器相关命令
	3种模式
		命令模式
		输入模式
		底线命令模式
	#3种模式功能职责 以及如何进行模式间的转换
    
    vim 文件路径
    i 进入输入模式
    G 到文件最后一行
    gg 文件的首行
    yy 复制
    nyy 复制多行
    p  粘贴当前行的下一行
    P  粘贴当前行的上一行
    u  撤销
    ctrl+r 反撤销
    dd 删除一行
    ndd 删除多行
    shift+zz 保存、退出
    :q! 不保存 强制退出
```
